</ParamField>

<ParamField body="phoneNumber" type="string" required>

</ParamField>

</ParamField>

</ParamField>

<ParamField body="accountName" type="string" required>

</ParamField>

<ParamField body="currency" type="string" required>

</ParamField>

</ParamField>
